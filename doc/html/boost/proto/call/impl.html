Struct template impl

boost::proto::call::impl

Synopsis

template<typename Expr, typename State, typename Data> 
struct impl :  proto::transform_impl<Expr, State, Data> {
 // types
typedef see-below result_type;

 // public member functions
result_type operator()(typename impl::expr_param, 
 typename impl::state_param, 
 typename impl::data_param) const;
};

Description

impl public types

  1. typedef see-below result_type;

    proto::call<T>::impl<Expr,State,Data>::result_type is computed as follows:
    proto::call<T>::impl<Expr,State,Data>::result_type 计算如下：

    • If T if of the form PrimitiveTransform or PrimitiveTransform(), then result_type is:
      如果 T 形如 PrimitiveTransform 或 PrimitiveTransform()，则 result_type 为： 

      typename boost::result_of<PrimitiveTransform(Expr, State, Data)>::type

    • If T is of the form PrimitiveTransform(T0), then result_type is:
      如果 T 形如 PrimitiveTransform(T0)，则 result_type 为： 

      typename boost::result_of<PrimitiveTransform(
       typename boost::result_of<when<_,T0>(Expr, State, Data)>::type,
       State, 
       Data
      )>::type

    • If T is of the form PrimitiveTransform(T0, T1), then result_type is:
      如果 T 形如 PrimitiveTransform(T0, T1)，则 result_type 为： 

      typename boost::result_of<PrimitiveTransform(
       typename boost::result_of<when<_,T0>(Expr, State, Data)>::type,
       typename boost::result_of<when<_,T1>(Expr, State, Data)>::type,
       Data
      )>::type

    • If T is of the form PrimitiveTransform(T0, T1, T2), then result_type is:
      如果 T 形如 PrimitiveTransform(T0, T1, T2)，则 result_type 为： 

      typename boost::result_of<PrimitiveTransform(
       typename boost::result_of<when<_,T0>(Expr, State, Data)>::type,
       typename boost::result_of<when<_,T1>(Expr, State, Data)>::type,
       typename boost::result_of<when<_,T2>(Expr, State, Data)>::type
      )>::type

    • If T is of the form PolymorphicFunctionObject(T0,...TN), then result_type is:
      如果 T 形如 PolymorphicFunctionObject(T0,...TN)，则 result_type 为： 

      typename boost::result_of<PolymorphicFunctionObject(
       typename boost::result_of<when<_,T0>(Expr, State, Data)>::type,
       ...
       typename boost::result_of<when<_,TN>(Expr, State, Data)>::type
      >::type

impl public member functions

  1. result_type operator()(typename impl::expr_param expr, 
     typename impl::state_param state, 
     typename impl::data_param data) const;

    proto::call<T>::impl<Expr,State,Data>::operator() behaves as follows:
    proto::call<T>::impl<Expr,State,Data>::operator() 的行为如下：

    • If T if of the form PrimitiveTransform or PrimitiveTransform(), then return
      如果 T 形如 PrimitiveTransform 或 PrimitiveTransform()，则返回 

      PrimitiveTransform()(expr, state, data)

    • If T is of the form PrimitiveTransform(T0), then return
      如果 T 形如 PrimitiveTransform(T0)，则返回 

      PrimitiveTransform()(
       when<_,T0>()(expr, state, data),
       state, 
       sata
      )

    • If T is of the form PrimitiveTransform(T0, T1), then return:
      如果 T 形如 PrimitiveTransform(T0, T1)，则返回 

      PrimitiveTransform()(
       when<_,T0>()(expr, state, data),
       when<_,T1>()(expr, state, data),
       Data
      )

    • If T is of the form PrimitiveTransform(T0, T1, T2), then return
      如果 T 形如 PrimitiveTransform(T0, T1, T2)，则返回 

      PrimitiveTransform()(
       when<_,T0>()(expr, state, data),
       when<_,T1>()(expr, state, data),
       when<_,T2>()(expr, state, data)
      )

    • If T is of the form PolymorphicFunctionObject(T0,...TN), then return:
      如果 T 形如 PolymorphicFunctionObject(T0,...TN)，则返回 

      PolymorphicFunctionObject()(
       when<_,T0>()(expr, state, data),
       ...
       when<_,TN>()(expr, state, data)
      )
