bind

摘要

template< 
      typename F
    >
struct bind0
{
    // unspecified
    // ...
};

template< 
      typename F, typename A1
    >
struct bind1
{
    // unspecified
    // ...
};

...

template< 
      typename F, typename A1,... typename An
    >
struct bindn
{
    // unspecified
    // ...
};

template< 
      typename F
    , typename A1 = unspecified
    ...
    , typename An = unspecified
    >
struct bind
{
    // unspecified
    // ...
};

描述

bind 是一个高阶基本模板，用于 元函数类 合成和参数绑定。本质上，它是 Boost.Bind 和 Boost.Lambda 库所提供的运行期功能在编译期中的对应物。

头文件

#include <boost/mpl/bind.hpp>

Model of

元函数类

参数

参数 要求 描述
F 元函数类 执行绑定的元函数类
A1,... An 任意类型 绑定的参数

表达式语义

对于任意 元函数类 f 以及任意类型 a1,... an:

typedef bind<f,a1,...an> g;
typedef bindn<f,a1,...an> g;
返回类型: 元函数类
语义:

等价于

struct g
{
    template< 
          typename U1 = unspecified
        ...
        , typename Un = unspecified
        >
    struct apply
        : apply_wrapn<
              typename h0<f,U1,...Un>::type 
            , typename h1<a1,U1,...Un>::type 
            ...
            , typename hn<an,U1,...Un>::type 
            >
    {
    };
};

其中 hk 等价于

template< typename X, typename U1,... typename Un > struct hk
    : apply_wrapn<X,U1,...Un>
{
};

如果 f 或 ak 是一个 绑定表达式 或 占位符，否则为

template< typename X, typename U1,... typename Un > struct hk
{
    typedef X type;
};

[注: 在 bind<f,a1,...an> 中第 n 次出现的 匿名占位符 将被替换为带数字的占位符 _n — 注完]

例子

struct f1
{
    template< typename T1 > struct apply
    {
        typedef T1 type;
    };
};

struct f5
{
    template< typename T1, typename T2, typename T3, typename T4, typename T5 >
    struct apply
    {
        typedef T5 type;
    };
};

typedef apply_wrap1< 
      bind1<f1,_1>
    , int 
    >::type r11;

typedef apply_wrap5< 
      bind1<f1,_5>
    , void,void,void,void,int 
    >::type r12;

BOOST_MPL_ASSERT(( is_same<r11,int> ));
BOOST_MPL_ASSERT(( is_same<r12,int> ));

typedef apply_wrap5< 
      bind5<f5,_1,_2,_3,_4,_5>
    , void,void,void,void,int 
    >::type r51;

typedef apply_wrap5<
      bind5<f5,_5,_4,_3,_2,_1>
    , int,void,void,void,void
    >::type r52;

BOOST_MPL_ASSERT(( is_same<r51,int> ));
BOOST_MPL_ASSERT(( is_same<r52,int> ));
